web 协议：
Cerberus与前端通信协议：
class FrontEndingCommunicationProtocol<T>
包含字段
{
	"condition":{
	　　"operateType":操作类型,string:"operateType","tableType","status","length","offset","ok","fail"
	　　"tableType":将要被操作的表名(如流水信息，用来表示流水信息的具体类型："package","item"),string
	　　"status":返回状态，表示操作是否成功,string
	　　"offset":偏移量,integer
		"length":长度,integer
	},
	"rows":[
		多组数据，根据T的类型来确定
	]
}
